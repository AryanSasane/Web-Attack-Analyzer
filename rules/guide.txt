TRAVERSAL.PY
#A Traversal attack tricks the server into going outside that folder using ../ (which means "go up one folder" in Linux) to a real Linux file containing all system usernames. The attacker just read a file they were never supposed to access.

# {2,} means repeat this 2 or more times so it catches the same pattern 2 or more times as well
# \ is also a regex char so it needs escaping by adding a \, thus \\

    r"(\.\./){2,}",             #detects the classical traversal pattern ../../
    r"(\.\.\\){2,}",            #detects windows path using backslash ..\\
    r"(?i)%2e%2e%2f",           #detects URL encoded version of ../, URL encoding : %2e -> . & %2f -> /
    r"(?i)/etc/passwd",         #directly matches if someone is trying to access the Linux User accts file, no regex, if this appears defo attack
    r"(?i)/etc/shadow",         #directly matches shadow which holds a hashed pwd(worse than pwd)
    r"(?i)/windows/system32",   #detects attempt to acc windows sys folder - equi to attack on windows server





XSS.PY
#XSS is when an attacker injects malicious JavaScript into a webpage that other users then load in their browsers.

# .*? means "any characters, any amounts", the ? makes it lazy, stopping at the first > it finds
# on\w+ matches on flwed by an wrd characters, catching onerror, onclick, onmouseover, etc
# \s*=\s* matches = with optional spaces around it
# ['\"].*?['\"] matches anything inside quotes
# . means any character in regex, so we avoid that using \ thus \. means actual dot

    r"(?i)<script.*?>",                 #detects opening script tags like <script> or <script type = ...>
    r"(?i)</script>",                   #detects closing script tags
    r"(?i)on\w+\s*=\s*['\"].*?['\"]",   #detects event handler injections like onerror="alert(1)" or onclick="steal()"
    r"(?i)javascript\s*:",              #detects javascript protocol injectns like href="javascript:alert(1)", a classic XSS vector in links
    r"(?i)eval\s*\(",                   #detects eval calls, used to exec arbitrary JavaScript code
    r"(?i)document\.cookie",            #detects attempts to steal cookies
